CREATE TABLE Users (
    username VARCHAR(50) NOT NULL,
    email VARCHAR(50),
    password VARCHAR(50),
    PRIMARY KEY (username) 
); 

CREATE TABLE Interests(
    username VARCHAR(50) NOT NULL, 
    interest_name VARCHAR(50) DEFAULT 'Random',
    PRIMARY KEY (username, interest_name),
    FOREIGN KEY (username) REFERENCES Users (username)
    ON DELETE CASCADE
);

CREATE TABLE Languages (
    username VARCHAR(50) NOT NULL,
    language VARCHAR(50) DEFAULT 'English',
    PRIMARY KEY (username, language),
    FOREIGN KEY (username) REFERENCES Users (username)
    ON DELETE CASCADE
); 

CREATE TABLE Penpals (
    pen_id INT NOT NULL AUTO_INCREMENT,
    user1 VARCHAR(50) NOT NULL,
    user2 VARCHAR(50) NOT NULL,
    PRIMARY KEY (pen_id),
    FOREIGN KEY (user1) REFERENCES Users (username),
    FOREIGN KEY (user2) REFERENCES Users (username)
    ON DELETE CASCADE
); 

CREATE TABLE Messages (
    message_id  INT NOT NULL AUTO_INCREMENT,
    pen_id INT NOT NULL,
    content VARCHAR(50),
    sender VARCHAR(50),
    whens DATETIME DEFAULT CURRENT_TIMESTAMP,  
    PRIMARY KEY (message_id),
    FOREIGN KEY (pen_id) REFERENCES Penpals (pen_id)
    ON DELETE CASCADE
);
